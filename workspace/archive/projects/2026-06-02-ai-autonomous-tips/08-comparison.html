function routeTask(task) {
  // 确定性分流决策树
  if (task.hasExplicitRule) {
    return 'deterministic'; // if/else 路径
  } else if (task.requiresJudgment) {
    return 'ai-assisted'; // AI 判断路径
  } else if (task.isSimilarTo(previousTask)) {
    return 'batch-process'; // 合并同类项
  } else {
    checkPreconditions(task); // 确认世界没变过
    return 'execute-first'; // 干了再说
  }
}
