// 供应商评分计算函数
function calculateScore(metrics) {
  const weights = {
    capacity: 0.35,
    quality: 0.30,
    compliance: 0.20,
    cost: 0.15
  };
  return Object.keys(weights).reduce((sum, key) =>
    sum + metrics[key] * weights[key], 0
  );
}

// 返回综合评分 (0-100)
const score = calculateScore({
  capacity: 92, quality: 88, compliance: 78, cost: 85
});
