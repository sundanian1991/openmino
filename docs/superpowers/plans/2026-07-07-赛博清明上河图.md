- [ ] **Step 1: 在 `</body>` 前的 `<script>` 顶部加入密度生成函数**

找到 `const scroller=document.getElementById('scroller');` 这一行，在它**之前**插入通用生成函数：

```js
// ===== 程序化密度层通用工具 =====
const NEON=['#00f0ff','#ff2d95','#ffd23f','#b14dff'];
function rnd(min,max){return min+Math.random()*(max-min)}
function pick(arr){return arr[Math.floor(Math.random()*arr.length)]}

function spawnPeople(container,count,opts){
  const {yMin,yMax,sMin,sMax,glow}=opts;
  for(let i=0;i<count;i++){
    const s=rnd(sMin,sMax);
    const el=document.createElement('div');
    el.style.cssText=`position:absolute;left:${rnd(0,100)}%;top:${rnd(yMin,yMax)}%;width:${s}px;height:${s*2.2}px;background:#0a0408;box-shadow:0 0 2px ${pick(glow||NEON)};border-radius:40% 40% 30% 30%;opacity:${rnd(.6,.95)}`;
    container.appendChild(el);
  }
}
function spawnSigns(container,count,opts){
  const {yMin,yMax,wMin,wMax,hMin,hMax}=opts;
  for(let i=0;i<count;i++){
    const c=pick(NEON);
    const el=document.createElement('div');
    el.style.cssText=`position:absolute;left:${rnd(0,100)}%;top:${rnd(yMin,yMax)}%;width:${rnd(wMin,wMax)}px;height:${rnd(hMin,hMax)}px;background:${c};opacity:${rnd(.25,.6)};box-shadow:0 0 ${rnd(4,12)}px ${c};border-radius:1px`;
    container.appendChild(el);
  }
}
function spawnLanterns(container,count,opts){
  const {yMin,yMax}=opts;
  for(let i=0;i<count;i++){
    const c=pick(NEON);
    const el=document.createElement('div');
    el.style.cssText=`position:absolute;left:${rnd(0,100)}%;top:${rnd(yMin,yMax)}%;width:6px;height:6px;background:${c};border-radius:50%;box-shadow:0 0 14px ${c},0 0 28px ${c};opacity:${rnd(.6,.9)}`;
    container.appendChild(el);
  }
}
```

**目标**：给三个画眼加专属 Canvas 动效（雾涌/水波动/光柱倾泻），IntersectionObserver 控制只在视口内运行。这是性能护栏。

- [ ] **Step 1: 在 `<script>` 末尾加入画眼动效引擎**

```js
// ===== 画眼 Canvas 动效（IntersectionObserver 控启停） =====
const focalAnims={1:null,2:null,3:null};
function startFocal(canvas,focalId){
  const ctx=canvas.getContext('2d');
  function resize(){const r=canvas.parentElement.getBoundingClientRect();canvas.width=r.width;canvas.height=r.height}
  resize();
  let t=0,raf;
  const particles=[];
  for(let i=0;i<60;i++)particles.push({x:Math.random(),y:Math.random(),vy:0.0003+Math.random()*0.001,vx:(Math.random()-.5)*0.0005,r:0.5+Math.random()*1.8,c:pick(['#00f0ff','#ff2d95','#ffd23f','#b14dff']),a:0.15+Math.random()*0.5});

  function draw(){
    ctx.clearRect(0,0,canvas.width,canvas.height);
    t+=0.016;
    const W=canvas.width,H=canvas.height;

    if(focalId===1){
      // 雾涌 + 塔光脉冲
      const pulse=0.5+0.5*Math.sin(t*0.8);
      ctx.fillStyle=`rgba(0,240,255,${0.05+0.05*pulse})`;
      ctx.fillRect(0,H*0.7,W,H*0.3);
    }else if(focalId===2){
      // 水面强倒影波动 + 桥体粒子
      const wave=Math.sin(t*2);
      ctx.fillStyle=`rgba(0,240,255,${0.04+0.03*wave})`;
      ctx.fillRect(0,H*0.8,W,H*0.2);
      ctx.fillStyle=`rgba(255,45,149,${0.03+0.02*Math.sin(t*2.5)})`;
      ctx.fillRect(0,H*0.85,W,H*0.15);
    }else if(focalId===3){
      // 门洞光柱倾泻 + 扫描线
      const grad=ctx.createLinearGradient(W*0.5,H*0.2,W*0.5,H);
      grad.addColorStop(0,'rgba(255,210,63,0.12)');
      grad.addColorStop(1,'rgba(255,210,63,0)');
      ctx.fillStyle=grad;
      ctx.beginPath();
      ctx.moveTo(W*0.3,0);ctx.lineTo(W*0.7,0);ctx.lineTo(W*0.85,H);ctx.lineTo(W*0.15,H);
      ctx.closePath();ctx.fill();
      // 安检扫描线
      const scanY=(t*60)%H;
      ctx.strokeStyle='rgba(0,240,255,0.3)';ctx.lineWidth=1;
      ctx.beginPath();ctx.moveTo(0,scanY);ctx.lineTo(W,scanY);ctx.stroke();
    }

    // 通用粒子
    particles.forEach(p=>{
      p.y+=p.vy;p.x+=p.vx;
      if(p.y>1)p.y=0;if(p.x<0)p.x=1;if(p.x>1)p.x=0;
      ctx.fillStyle=p.c;ctx.globalAlpha=p.a;
      ctx.beginPath();ctx.arc(p.x*W,p.y*H,p.r,0,Math.PI*2);ctx.fill();
    });
    ctx.globalAlpha=1;
    raf=requestAnimationFrame(draw);
  }
  draw();
  focalAnims[focalId]={raf,resize};
}
function stopFocal(focalId){
  const a=focalAnims[focalId];
  if(a&&a.raf){cancelAnimationFrame(a.raf);a.raf=null}
}

const focals=document.querySelectorAll('.focal');
const io=new IntersectionObserver((entries)=>{
  entries.forEach(e=>{
    const canvas=e.target.querySelector('.focal-canvas');
    const id=e.target.dataset.focal;
    if(e.isIntersecting){startFocal(canvas,id)}else{stopFocal(id)}
  });
},{threshold:0.15});
focals.forEach(f=>io.observe(f));
window.addEventListener('resize',()=>{Object.values(focalAnims).forEach(a=>{if(a&&a.resize)a.resize()})});
```

常见瓶颈与对策：
- 粒子太多 → 减少画眼粒子数（60→40）
- 元素太多 → 减少第三幕人群（200→150）
