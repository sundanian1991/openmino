## 2. 数据结构

### 2.1 RAW数据格式（108条记录）

```javascript
{
  id: 1,                    // 序号
  site: "颍州站点",         // 职场名称
  company: "衍生",          // 供应商全称
  short: "衍生",            // 供应商简称（用于分组）
  province: "安徽",         // 省份
  city: "阜阳",             // 城市
  district: "颍州区",       // 区县
  region: "华东",           // 区域
  tier: "三线",             // 城市能级
  product: "企业主贷",      // 产品线
  bizLine: "企金",          // 业务线
  mode: "人工",             // 运营模式（人工/人机）
  m1: 12,                   // 1月人力
  m2: 8,                    // 2月人力
  m3: 13                    // 3月人力
}
```

### 2.2 聚合引擎

#### mergeWorkplaces(data)
- **功能**：按（short + site）合并职场
- **返回**：职场数组，每个元素包含：
  - short, site, province, city, district, region, tier
  - products: []（产品线列表）
  - modes: []（运营模式列表）
  - m1, m2, m3（合并后人力）
  - headcount: 当前月人力
  - trend: m3 - m1（趋势）

#### mergeSuppliers(data)
- **功能**：按short合并供应商
- **返回**：供应商数组，每个元素包含：
  - short, company
  - products: []（产品线列表）
  - siteCount: 站点数
  - m1, m2, m3
  - headcount: 当前月人力
  - trend: m3 - m1

#### mergeProducts(data)
- **功能**：按product合并产品线
- **返回**：产品线数组，每个元素包含：
  - product, bizLine
  - supplierCount: 供应商数
  - siteCount: 站点数
  - modes: {人工: 0, 人机: 0}
  - provinces: Set()
  - m1, m2, m3
  - headcount: 当前月人力

### 2.3 筛选器逻辑

```javascript
let currentMonth = 3;        // 当前月份（1/2/3）
let filterProvince = '';     // 省份筛选
let filterSupplier = '';     // 供应商筛选
let filterProduct = '';      // 产品线筛选

function getFiltered() {
  return RAW.filter(r => {
    if (filterProvince && r.province !== filterProvince) return false;
    if (filterSupplier && r.short !== filterSupplier) return false;
    if (filterProduct && r.product !== filterProduct) return false;
    return true;
  });
}
```

## 3. 10大风险维度

### 3.1 风险维度定义

| # | 风险维度 | 对象 | 定义 | 阈值 | 风险等级 |
|---|---------|------|------|------|----------|
| 1 | 微小站点 | 站点 | headcount < 20 | 0 | count>0 → 高风险 |
| 2 | 超大站点 | 站点 | headcount > 100 | 0 | count>0 → 高风险 |
| 3 | 供应商集中度 | 供应商 | 在单一项目的人力占供应商总人力 | >30% | >30% → 高风险 |
| 4 | 供应链单薄 | 项目 | 供应商数量 | ≤2家 | count>0 → 高风险 |
| 5 | 地理集中度 | 城市 | 同城市站点数 | >2 | count>2 → 高风险 |
| 6 | 项目集中度 | 项目 | TOP3供应商占该项目总人力 | >80% | >80% → 高风险 |
| 7 | 连续下降 | 供应商/站点 | 月度环比连续下降 | ≥2个月 | count>0 → 高风险 |
| 8 | 业务灵活性低 | 供应商 | 只做1条产品线 | 0 | count>0 → 中风险 |
| 9 | 业务分散风险 | 供应商 | 做>5条产品线 | 0 | count>0 → 中风险 |
| 10 | 站点波动 | 站点 | 月度人力波动 >20% | 0 | count>0 → 中风险 |

### 3.2 计算函数

```javascript
// 1. 微小站点
function calcTinySiteRisk(workplaces) {
  const tiny = workplaces.filter(w => w.headcount < 20);
  return { count: tiny.length, items: tiny };
}

// 2. 超大站点
function calcHugeSiteRisk(workplaces) {
  const huge = workplaces.filter(w => w.headcount > 100);
  return { count: huge.length, items: huge };
}

// 3. 供应商集中度（项目维度）
function calcSupplierConcentrationRisk(data, suppliers) {
  const riskySuppliers = [];
  suppliers.forEach(s => {
    const supData = data.filter(r => r.short === s.short);
    const productMap = {};
    let total = 0;
    supData.forEach(r => {
      const val = getVal(r, currentMonth);
      productMap[r.product] = (productMap[r.product] || 0) + val;
      total += val;
    });
    if (total > 0) {
      Object.entries(productMap).forEach(([product, val]) => {
        const pct = val / total * 100;
        if (pct > 30) {
          riskySuppliers.push({ supplier: s.short, product, pct: Math.round(pct), val, total });
        }
      });
    }
  });
  return { count: riskySuppliers.length, items: riskySuppliers, threshold: 30 };
}

// 4. 供应链单薄
function calcThinSupplyRisk(products) {
  const thin = products.filter(p => p.supplierCount <= 2);
  return { count: thin.length, items: thin };
}

// 5. 地理集中度（同城市站点数 > 2）
function calcGeoConcentrationRisk(workplaces) {
  const cityMap = {};
  workplaces.forEach(w => {
    cityMap[w.city] = (cityMap[w.city] || 0) + 1;
  });
  const riskyCities = Object.entries(cityMap).filter(([, count]) => count > 2);
  const items = riskyCities.map(([city, count]) => ({ city, count, sites: workplaces.filter(w => w.city === city) }));
  return { count: riskyCities.length, items, threshold: 2 };
}

// 6. 项目集中度（TOP3供应商占比 > 80%）
function calcProjectConcentrationRisk(products, data) {
  const riskyProducts = [];
  products.forEach(p => {
    const pData = data.filter(r => r.product === p.product);
    const supMap = {};
    let total = 0;
    pData.forEach(r => {
      const val = getVal(r, currentMonth);
      supMap[r.short] = (supMap[r.short] || 0) + val;
      total += val;
    });
    if (total > 0) {
      const sorted = Object.entries(supMap).sort((a, b) => b[1] - a[1]);
      const top3 = sorted.slice(0, 3);
      const top3Sum = top3.reduce((s, [, v]) => s + v, 0);
      const pct = top3Sum / total * 100;
      if (pct > 80) {
        riskyProducts.push({ product: p.product, pct: Math.round(pct), top3: top3.map(([s]) => s), total });
      }
    }
  });
  return { count: riskyProducts.length, items: riskyProducts, threshold: 80 };
}

// 7. 连续下降
function calcContinuousDeclineRisk(entities) {
  // entities可以是workplaces或suppliers
  const declining = entities.filter(e => {
    const trend1 = getVal(e, 2) - getVal(e, 1);
    const trend2 = getVal(e, 3) - getVal(e, 2);
    return trend1 < 0 && trend2 < 0;
  });
  return { count: declining.length, items: declining };
}

// 8. 业务灵活性低
function calcLowFlexibilityRisk(suppliers) {
  const inflexible = suppliers.filter(s => s.products.length === 1);
  return { count: inflexible.length, items: inflexible };
}

// 9. 业务分散风险
function calcOverDiversifiedRisk(suppliers) {
  const diversified = suppliers.filter(s => s.products.length > 5);
  return { count: diversified.length, items: diversified };
}

// 10. 站点波动
function calcSiteFluctuationRisk(workplaces) {
  const fluctuating = [];
  workplaces.forEach(w => {
    const m1 = getVal(w, 1);
    const m2 = getVal(w, 2);
    const m3 = getVal(w, 3);
    // 计算月度波动
    const vol1 = m1 > 0 ? Math.abs((m2 - m1) / m1 * 100) : 0;
    const vol2 = m2 > 0 ? Math.abs((m3 - m2) / m2 * 100) : 0;
    if (vol1 > 20 || vol2 > 20) {
      fluctuating.push({ site: w.site, supplier: w.short, vol1: Math.round(vol1), vol2: Math.round(vol2), m1, m2, m3 });
    }
  });
  return { count: fluctuating.length, items: fluctuating, threshold: 20 };
}
```

### 5.2 供应商风险评级算法

```javascript
function calcSupplierRiskRating(supplier, data, workplaces) {
  let riskScore = 0;
  let riskCount = 0;

  // 1. 微小站点：每个+10分
  const tinySites = workplaces.filter(w => w.short === supplier.short && w.headcount < 20);
  riskScore += tinySites.length * 10;
  if (tinySites.length > 0) riskCount++;

  // 2. 超大站点：每个+15分
  const hugeSites = workplaces.filter(w => w.short === supplier.short && w.headcount > 100);
  riskScore += hugeSites.length * 15;
  if (hugeSites.length > 0) riskCount++;

  // 3. 供应商集中度：>30% → +25分
  const concRisk = calcSupplierConcentrationRiskForSupplier(supplier, data);
  if (concRisk > 30) { riskScore += 25; riskCount++; }

  // 4. 业务灵活性低：+20分
  if (supplier.products.length === 1) { riskScore += 20; riskCount++; }

  // 5. 连续下降：+30分
  const isDeclining = (getVal(supplier, 2) - getVal(supplier, 1) < 0) && (getVal(supplier, 3) - getVal(supplier, 2) < 0);
  if (isDeclining) { riskScore += 30; riskCount++; }

  // 评级
  if (riskScore >= 50 || riskCount >= 3) return '高风险';
  if (riskScore >= 25 || riskCount >= 1) return '中风险';
  return '低风险';
}
```

## 9. 交付物格式

### 9.1 技术要求

- **单文件HTML**：所有代码（HTML+CSS+JS）在一个文件中
- **零依赖**：除ECharts 5.5.0外不依赖其他库
- **ECharts版本**：5.5.0（CDN：https://cdn.jsdelivr.net/npm/echarts@5.5.0/dist/echarts.min.js）
- **浏览器兼容**：Chrome 90+、Safari 14+、Edge 90+

```
workspace/output/
```
