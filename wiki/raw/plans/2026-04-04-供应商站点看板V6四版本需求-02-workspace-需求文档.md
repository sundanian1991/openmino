## 2. 数据结构

### 2.1 RAW数据格式（108条记录）

```javascript
{
  id: 1,                    // 序号
  site: "颍州站点",         // 职场名称
  company: "衍生",          // 供应商全称
  short: "衍生",            // 供应商简称（用于分组）
  province: "安徽",         // 省份
  city: "阜阳",             // 城市
  district: "颍州区",       // 区县
  region: "华东",           // 区域
  tier: "三线",             // 城市能级
  product: "企业主贷",      // 产品线
  bizLine: "企金",          // 业务线
  mode: "人工",             // 运营模式（人工/人机）
  m1: 12,                   // 1月人力
  m2: 8,                    // 2月人力
  m3: 13                    // 3月人力
}
```

### 2.2 聚合引擎

#### mergeWorkplaces(data)
- **功能**：按（short + site）合并职场
- **返回**：职场数组，每个元素包含：
  - short, site, province, city, district, region, tier
  - products: []（产品线列表）
  - modes: []（运营模式列表）
  - m1, m2, m3（合并后人力）
  - headcount: 当前月人力
  - trend: m3 - m1（趋势）

#### mergeSuppliers(data)
- **功能**：按short合并供应商
- **返回**：供应商数组，每个元素包含：
  - short, company
  - products: []（产品线列表）
  - siteCount: 站点数
  - m1, m2, m3
  - headcount: 当前月人力
  - trend: m3 - m1

#### mergeProducts(data)
- **功能**：按product合并产品线
- **返回**：产品线数组，每个元素包含：
  - product, bizLine
  - supplierCount: 供应商数
  - siteCount: 站点数
  - modes: {人工: 0, 人机: 0}
  - provinces: Set()
  - m1, m2, m3
  - headcount: 当前月人力

### 2.3 筛选器逻辑

```javascript
let currentMonth = 3;        // 当前月份（1/2/3）
let filterProvince = '';     // 省份筛选
let filterSupplier = '';     // 供应商筛选
let filterProduct = '';      // 产品线筛选

function getFiltered() {
  return RAW.filter(r => {
    if (filterProvince && r.province !== filterProvince) return false;
    if (filterSupplier && r.short !== filterSupplier) return false;
    if (filterProduct && r.product !== filterProduct) return false;
    return true;
  });
}
```

## 3. 8大风险维度

### 3.1 风险维度定义

| # | 风险维度 | 定义 | 计算逻辑 | 阈值 | 风险等级判定 |
|---|----------|------|----------|------|--------------|
| 1 | 微小站点 | headcount < 20 | `workplaces.filter(w => w.headcount < 20)` | 0 | count > 0 → 高风险 |
| 2 | 超大站点 | headcount > 100 | `workplaces.filter(w => w.headcount > 100)` | 0 | count > 0 → 高风险 |
| 3 | 单产品供应 | 某产品线供应商数 < 3 | `products.filter(p => p.supplierCount < 3)` | 0 | count > 0 → 中风险 |
| 4 | 单产品线供应商 | 供应商只服务1条产品线 | `suppliers.filter(s => s.products.length === 1)` | `count > suppliers.length * 0.3` | 超阈值 → 中风险 |
| 5 | 供应链单薄 | 同产品线供应商数 < 3 | `products.filter(p => p.supplierCount < 3)` | 0 | count > 0 → 中风险 |
| 6 | 最大省份集中度 | 最大省占比 | `max(prov人力总和) / total * 100` | 40% | >50%高风险，>40%中风险 |
| 7 | **地理集中度** | **同城市站点数 > 2** | **按city字段统计站点数** | **0** | **count>2高风险，count>0中风险** |
| 8 | **项目单一风险** | **最大单一项目占比 > 70%** | **供应商各产品线中最大占比** | **70%** | **超过70%高风险** |

### 3.2 计算函数签名

```javascript
// 1. 微小站点
function calcTinySiteRisk(workplaces) {
  const tiny = workplaces.filter(w => w.headcount < 20);
  return { count: tiny.length, items: tiny, threshold: 0, max: Math.max(5, Math.ceil(workplaces.length * 0.15)) };
}

// 2. 超大站点
function calcHugeSiteRisk(workplaces) {
  const huge = workplaces.filter(w => w.headcount > 100);
  return { count: huge.length, items: huge, threshold: 0, max: Math.max(3, Math.ceil(workplaces.length * 0.1)) };
}

// 3. 单产品供应
function calcSingleProductRisk(products) {
  const single = products.filter(p => p.supplierCount < 3);
  return { count: single.length, items: single, threshold: 0, max: Math.max(2, Math.ceil(products.length * 0.1)) };
}

// 4. 单产品线供应商
function calcSingleBizSupplierRisk(suppliers) {
  const single = suppliers.filter(s => s.products.length === 1);
  return { count: single.length, items: single, threshold: Math.floor(suppliers.length * 0.3), max: Math.floor(suppliers.length * 0.6) };
}

// 5. 供应链单薄
function calcThinSupplyRisk(products) {
  const thin = products.filter(p => p.supplierCount < 3);
  return { count: thin.length, items: thin, threshold: 0, max: Math.max(2, Math.ceil(products.length * 0.1)) };
}

// 6. 最大省份集中度
function calcProvConcentrationRisk(workplaces) {
  const provMap = {};
  workplaces.forEach(w => { provMap[w.province] = (provMap[w.province] || 0) + w.headcount; });
  const total = workplaces.reduce((s, w) => s + w.headcount, 0);
  if (!total) return { count: 0, pct: 0, threshold: 40, max: 70 };
  const maxProv = Math.max(...Object.values(provMap));
  const maxPct = Math.round(maxProv / total * 100);
  return { count: maxPct, pct: maxPct, threshold: 40, max: 70 };
}

// 7. 地理集中度（同城市站点数 > 2）
function calcGeoConcentrationRisk(workplaces) {
  const cityMap = {};
  workplaces.forEach(w => {
    cityMap[w.city] = (cityMap[w.city] || 0) + 1;
  });
  const riskyCities = Object.entries(cityMap).filter(([, count]) => count > 2);
  const maxCount = Math.max(...Object.values(cityMap), 0);
  const items = riskyCities.map(([city, count]) => ({ city, count, sites: workplaces.filter(w => w.city === city) }));
  return { count: riskyCities.length, maxCount, items, threshold: 0, max: 5 };
}

// 8. 项目单一风险（最大单一项目占比 > 70%）
function calcProjectConcentrationRisk(data, suppliers) {
  const riskySuppliers = [];
  suppliers.forEach(s => {
    const supData = data.filter(r => r.short === s.short);
    const productMap = {};
    let total = 0;
    supData.forEach(r => {
      const val = getVal(r, currentMonth);
      productMap[r.product] = (productMap[r.product] || 0) + val;
      total += val;
    });
    if (total > 0) {
      const maxPct = Math.max(...Object.values(productMap).map(v => v / total * 100));
      if (maxPct > 70) {
        const maxProduct = Object.entries(productMap).sort((a, b) => b[1] - a[1])[0][0];
        riskySuppliers.push({ supplier: s.short, product: maxProduct, pct: Math.round(maxPct), val: productMap[maxProduct], total });
      }
    }
  });
  return { count: riskySuppliers.length, items: riskySuppliers, threshold: 70, max: 100 };
}
```

### 5.2 供应商风险评级算法

**短期方案（简单加权）**：
```javascript
function calcSupplierRiskRating(supplier, data, workplaces) {
  let riskScore = 0;

  // 1. 微小站点：每个+10分
  const tinySites = workplaces.filter(w => w.short === supplier.short && w.headcount < 20);
  riskScore += tinySites.length * 10;

  // 2. 超大站点：每个+15分
  const hugeSites = workplaces.filter(w => w.short === supplier.short && w.headcount > 100);
  riskScore += hugeSites.length * 15;

  // 3. 单产品线：+20分
  if (supplier.products.length === 1) riskScore += 20;

  // 4. 项目集中：最大单一项目>70% → +30分
  const projRisk = calcProjectConcentrationRiskForSupplier(supplier, data);
  if (projRisk > 70) riskScore += 30;

  // 5. 地理集中：同城市>2个站点 → +15分
  const cityMap = {};
  workplaces.filter(w => w.short === supplier.short).forEach(w => {
    cityMap[w.city] = (cityMap[w.city] || 0) + 1;
  });
  const maxCityCount = Math.max(...Object.values(cityMap), 0);
  if (maxCityCount > 2) riskScore += 15;

  // 评级
  if (riskScore >= 50) return '高风险';
  if (riskScore >= 25) return '中风险';
  return '低风险';
}
```
